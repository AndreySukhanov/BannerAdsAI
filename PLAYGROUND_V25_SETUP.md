    global pipe
    print("üé® –ó–∞–≥—Ä—É–∂–∞–µ–º Playground v2.5...")
    
    device = "cuda" if torch.cuda.is_available() else "cpu"

        torch_dtype=dtype,
        variant="fp16" if device == "cuda" else None

    print("‚úÖ Playground v2.5 –≥–æ—Ç–æ–≤–∞ –∫ —Ä–∞–±–æ—Ç–µ!")

async def generate_advertisement(
    prompt: str,
    width: int = 1024,
    height: int = 1024,
    steps: int = 50,
    guidance_scale: float = 3.0
):
    """
    –ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Ä–µ–∫–ª–∞–º–Ω–æ–≥–æ –±–∞–Ω–Ω–µ—Ä–∞ —á–µ—Ä–µ–∑ Playground v2.5
    """

        # –û–ø—Ç–∏–º–∏–∑–∏—Ä—É–µ–º –ø—Ä–æ–º–ø—Ç –¥–ª—è —Ä–µ–∫–ª–∞–º—ã

            "model": "playground-v2.5",
            "width": width,
            "height": height,

        }

      height: options.height || 1024,

      imageUrl: response.data.image,
      model: 'playground-v2.5',
      width: response.data.width,
      height: response.data.height
    };
  } catch (error) {
    throw new Error(`Playground generation failed: ${error.message}`);

}
